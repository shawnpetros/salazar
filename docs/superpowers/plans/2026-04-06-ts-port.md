# Salazar TypeScript Port — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port Salazar from a Python+Node polyglot to a single TypeScript npm package, eliminating the subprocess IPC layer and simplifying the architecture.

**Architecture:** Single npm package. Engine (orchestrator + agents) emits typed events via EventEmitter. TUI (Ink/React) subscribes directly — no log files, no subprocess spawning. CLI routes to either headless mode or TUI. `@anthropic-ai/claude-agent-sdk` replaces Python's `claude-agent-sdk` for programmatic Claude Code sessions.

**Tech Stack:** TypeScript, `@anthropic-ai/claude-agent-sdk`, `better-sqlite3`, `simple-git`, Ink 5, React 18, meow, tsup, vitest

**What's cut (intentionally deferred):**
- Brownfield mode (explorer, hardening, regression)
- Multi-orchestrator (architect agent, parallel services)
- Dashboard webhooks (remote push)
- Python project validator detection
- Environment variable model overrides

import { EventEmitter } from "node:events";
import type {
  SessionStartEvent, PlannerCompleteEvent, FeatureStartEvent,
  FeatureCompleteEvent, ValidatorResultEvent, EvaluatorResultEvent,
  CostUpdateEvent, SessionCompleteEvent, SessionErrorEvent,
} from "./types.js";

export interface EngineEventMap {
  sessionStart: [SessionStartEvent];
  plannerComplete: [PlannerCompleteEvent];
  featureStart: [FeatureStartEvent];
  featureComplete: [FeatureCompleteEvent];
  validatorResult: [ValidatorResultEvent];
  evaluatorResult: [EvaluatorResultEvent];
  costUpdate: [CostUpdateEvent];
  sessionComplete: [SessionCompleteEvent];
  sessionError: [SessionErrorEvent];
}

export class TypedEmitter extends EventEmitter {
  override emit<K extends keyof EngineEventMap>(
    event: K, ...args: EngineEventMap[K]
  ): boolean {
    return super.emit(event, ...args);
  }
  override on<K extends keyof EngineEventMap>(
    event: K, listener: (...args: EngineEventMap[K]) => void
  ): this {
    return super.on(event, listener as any);
  }
}

- [ ] **Step 1: Create `src/engine/security.ts`**

Port the bash allowlist from `salazar/salazar/security.py`. Same command set, same validation logic. The hook signature matches the TS Agent SDK's `HookCallback` type.

```typescript
const ALLOWED_COMMANDS = new Set([
  "ls", "cat", "head", "tail", "wc", "grep", "find",
  "cp", "mv", "mkdir", "chmod", "pwd", "echo", "printf",
  "npm", "npx", "node", "tsc", "eslint",
  "git", "touch", "rm", "sort", "uniq", "diff",
  "curl", "jq", "sed", "awk",
]);

function extractBaseCommands(command: string): string[] {
  return command.split(/[|;&]|&&|\|\|/)
    .map(p => p.trim())
    .filter(Boolean)
    .map(part => {
      const tokens = part.split(/\s+/);
      // Skip env var assignments
      for (const t of tokens) {
        if (t.includes("=") && t.indexOf("=") > 0) continue;
        return t.split("/").pop()!;
      }
      return tokens[0]?.split("/").pop() ?? "__EMPTY__";
    });
}

export async function bashSecurityHook(input: any): Promise<Record<string, unknown>> {
  const command = input?.input?.command ?? input?.tool_input?.command ?? "";
  if (!command) return { decision: "deny", reason: "Empty command" };
  for (const cmd of extractBaseCommands(command)) {
    if (!ALLOWED_COMMANDS.has(cmd)) {
      return { decision: "deny", reason: `Command '${cmd}' not in allowlist` };
    }
  }
  return { decision: "allow" };
}
```

import { readFileSync, existsSync } from "node:fs";
import { join } from "node:path";
import type { Feature, ProgressReport } from "../lib/types.js";

export function readProgress(workDir: string): ProgressReport | null {
  const path = join(workDir, "feature_list.json");
  if (!existsSync(path)) return null;
  const data = JSON.parse(readFileSync(path, "utf-8"));
  const items: Feature[] = Array.isArray(data) ? data : data.features ?? [];
  const passing = items.filter(f => f.passes).length;
  return {
    total: items.length,
    passing,
    items,
    percent: items.length > 0 ? (passing / items.length) * 100 : 0,
    isComplete: passing === items.length && items.length > 0,
  };
}

export function nextIncomplete(report: ProgressReport): Feature | null {
  return report.items.find(f => !f.passes) ?? null;
}

export function formatProgressHeader(report: ProgressReport): string {
  return `Progress: ${report.passing}/${report.total} features passing (${report.percent.toFixed(0)}%)\n`;
}
```

import { getDb } from "./storage.js";
import type { SalazarConfig } from "../lib/types.js";
import { execSync } from "node:child_process";
import crypto from "node:crypto";

const MAX_VALIDATOR_RETRIES = 3;
const MAX_EVALUATOR_RETRIES = 2;
const DELAY_BETWEEN_SESSIONS = 3000; // ms

export class Orchestrator extends TypedEmitter {
  private sessionId: string;
  private totalCost = 0;
  private costByAgent = { planner: 0, generator: 0, evaluator: 0 };
  private iteration = 0;
  private startTime = Date.now();

  constructor(
    private specPath: string,
    private outputDir: string,
    private config: SalazarConfig,
  ) {
    super();
    this.sessionId = crypto.randomUUID().replace(/-/g, "").slice(0, 12);
  }

  async run(): Promise<void> {
    this.emit("sessionStart", { type: "session_start", sessionId: this.sessionId, spec: this.specPath });

    // Phase 1: Planning
    const progress = readProgress(this.outputDir);
    if (!progress) {
      const planResult = await runPlanner(this.specPath, this.outputDir, this.config);
      if (!planResult.success) {
        this.emit("sessionError", { type: "session_error", error: "Planner failed" });
        return;
      }
      this.costByAgent.planner += planResult.costUsd;
      this.totalCost += planResult.costUsd;
      const p = readProgress(this.outputDir)!;
      this.emit("plannerComplete", { type: "planner_complete", features: p.total, durationMs: Date.now() - this.startTime });
    }

    // Phase 2: Build loop
    await this.featureLoop();
  }

  private async featureLoop(): Promise<void> {
    while (true) {
      const progress = readProgress(this.outputDir);
      if (!progress) { /* emit error */ return; }
      if (progress.isComplete) {
        this.emit("sessionComplete", {
          type: "session_complete",
          totalFeatures: progress.total,
          passing: progress.passing,
          durationMs: Date.now() - this.startTime,
          cost: this.totalCost,
        });
        return;
      }

      const feature = nextIncomplete(progress);
      if (!feature) { /* emit complete */ return; }

      this.iteration++;
      const featureStart = Date.now();

      this.emit("featureStart", {
        type: "feature_start",
        id: feature.id,
        iteration: this.iteration,
        done: progress.passing,
        total: progress.total,
        name: feature.description,
      });

      const skipEvaluator = feature.complexity === "setup" || feature.complexity === "simple";

      // Generator → Validators → Evaluator loop
      // (same retry logic as Python orchestrator)
      // ...abbreviated — full implementation follows the same pattern as
      // orchestrator.py lines 340-509

      await new Promise(r => setTimeout(r, DELAY_BETWEEN_SESSIONS));
    }
  }
}
```

import { useState, useEffect, useRef } from "react";
import { Orchestrator } from "../../engine/orchestrator.js";
import { loadConfig } from "../../lib/config.js";
import { OUTPUT_DIR } from "../../lib/paths.js";
import type { HarnessEvent } from "../../lib/types.js";

// Reuse the existing reducer and state shape from cli/src/hooks/use-harness.ts
// The reduceEvent function works identically — it just gets events from the
// engine directly instead of parsing them from log lines.

export function useEngine(specPath: string) {
  const [state, setState] = useState(/* initial state */);
  const engineRef = useRef<Orchestrator | null>(null);

  useEffect(() => {
    const config = loadConfig();
    const engine = new Orchestrator(specPath, OUTPUT_DIR, config);
    engineRef.current = engine;

    // Subscribe to ALL engine events, reduce into state
    const eventNames = [
      "sessionStart", "plannerComplete", "featureStart",
      "featureComplete", "validatorResult", "evaluatorResult",
      "costUpdate", "sessionComplete", "sessionError",
    ] as const;

    for (const name of eventNames) {
      engine.on(name, (event: any) => {
        setState(prev => reduceEvent(prev, event));
      });
    }

    // Start the engine (fire and forget — events drive the UI)
    engine.run().catch(err => {
      engine.emit("sessionError", { type: "session_error", error: String(err) });
    });

    return () => { /* cleanup if needed */ };
  }, [specPath]);

  return state;
}
